16/// Trait for service-specific configuration
17pub trait ServiceConfig: Clone + Send + Sync + 'static {
18    /// Loads configuration from system directories.
19    ///
20    /// # Errors
21    /// Returns `ServiceError::Config` if configuration loading or validation fails.
22    fn load() -> Result<Self, ServiceError>;
23    fn service_name(&self) -> &str;
24    fn time_interval(&self) -> u64;
25    fn log_file_path(&self) -> &str;
26}
27
28/// Action trait for extensible service functionality
29pub trait Action<C: ServiceConfig>: Send + Sync {
30    /// Executes the action with the given configuration.
31    ///
32    /// # Errors
33    /// Returns `ServiceError` if the action fails to execute.
34    fn execute(&self, config: &C) -> Result<(), ServiceError>;
35    fn name(&self) -> &str;
36}
37
38/// Service runner that manages and executes actions
39pub struct ServiceRunner<C: ServiceConfig> {
40    actions: Vec<Box<dyn Action<C>>>,
41}
42
43impl<C: ServiceConfig> Default for ServiceRunner<C> {
44    fn default() -> Self {
45        Self::new()
46    }
47}
48
49impl<C: ServiceConfig> ServiceRunner<C> {
50    #[must_use]
51    pub fn new() -> Self {
52        Self {
53            actions: Vec::new(),
54        }
55    }
56
57    #[must_use]
58    pub fn add_action(mut self, ACTION: Box<dyn Action<C>>) -> Self {
59        self.actions.push(ACTION);
60        self
61    }
62
63    /// Runs all registered actions.
64    ///
65    /// # Errors
66    /// Returns an error if any action fails to execute.
67    pub fn run(self) -> Result<(), Box<dyn std::error::Error>> {
68        let CONFIG = C::load()?;
69
70        // Check system resources before starting
71        if let Ok(config) = crate::service::config::load_config() {
72            monitoring::check_resources(&config)?;
73        }
74
75        let RUNNING = Arc::new(AtomicBool::new(true));
76
77        ctrlc::set_handler({
78            let RUNNING = Arc::clone(&RUNNING);
79            move || {
80                RUNNING.store(false, Ordering::SeqCst);
81            }
82        })?;
83
84        info!("{} starting...", CONFIG.service_name());
85
86        while RUNNING.load(Ordering::SeqCst) {
87            thread::sleep(Duration::from_secs(CONFIG.time_interval()));
88            if RUNNING.load(Ordering::SeqCst) {
89                for action in &self.actions {
90                    if let Err(e) = action.execute(&CONFIG) {
91                        log::error!("Action '{}' failed: {}", action.name(), e);
92                    }
93                }
94            }
95        }
96
97        info!("Service shutting down gracefully");
98        Ok(())
99    }
100}
