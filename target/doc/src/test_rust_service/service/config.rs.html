12/// Generic function to load config from system directories by filename
13fn load_config_by_filename(filename: &str) -> Result<String, ServiceError> {
14    let ALLOWED_CONFIGS = [
15        format!("/etc/rust-service-debug/{filename}"),
16        format!("/opt/rust-service-debug/{filename}"),
17        format!("/usr/local/etc/rust-service-debug/{filename}"),
18        format!("/etc/rust-service/{filename}"),
19        format!("/opt/rust-service/{filename}"),
20        format!("/usr/local/etc/rust-service/{filename}"),
21    ];
22
23    read_config_file(
24        &ALLOWED_CONFIGS
25            .iter()
26            .map(std::string::String::as_str)
27            .collect::<Vec<_>>(),
28    )
29}

31/// Validates that a configuration field value is within specified bounds.
32///
33/// # Errors
34/// Returns `ServiceError` if the value is outside the min/max range.
35pub fn validate_config_field<T: PartialOrd>(
36    value: &T,
37    min: &T,
38    max: &T,
39    name: &str,
40) -> Result<(), ServiceError> {
41    if value < min || value > max {
42        return Err(ServiceError::Config(format!("{name} out of range")));
43    }
44    Ok(())
45}

47/// Sanitizes a message by filtering allowed characters and checking length.
48///
49/// # Errors
50/// Returns `ServiceError` if the message is too long after sanitization.
51pub fn sanitize_message(MESSAGE: &str, MAX_LEN: usize) -> Result<String, ServiceError> {
52    let SANITIZED: String = MESSAGE
53        .chars()
54        .filter(|&C| C.is_ascii_alphanumeric() || C == ' ' || C == '.' || C == '-' || C == '_')
55        .take(MAX_LEN)
56        .collect();
57
58    if SANITIZED.is_empty() {
59        return Err(ServiceError::Config(
60            "Message cannot be empty after sanitization".to_string(),
61        ));
62    }
63    Ok(SANITIZED)
64}

66/// Reads configuration from the first available file in the provided paths.
67///
68/// # Errors
69/// Returns `ServiceError` if no configuration file is found or readable.
70pub fn read_config_file(config_paths: &[&str]) -> Result<String, ServiceError> {
71    let CONFIG_PATH = config_paths
72        .iter()
73        .find(|&&path| Path::new(path).exists())
74        .ok_or_else(|| ServiceError::Config("No valid config file found".to_string()))?;
75
76    // Open file and check permissions on file descriptor to prevent race conditions
77    let FILE = fs::File::open(CONFIG_PATH).map_err(|E| {
78        ServiceError::Config(format!("Failed to open config file {CONFIG_PATH}: {E}"))
79    })?;
80
81    #[cfg(unix)]
82    {
83        use std::os::unix::fs::MetadataExt;
84        let METADATA = FILE.metadata().map_err(|E| {
85            ServiceError::Config(format!(
86                "Cannot read config file metadata for {CONFIG_PATH}: {E}"
87            ))
88        })?;
89
90        if METADATA.mode() & 0o022 != 0 {
91            return Err(ServiceError::Config(
92                "Config file has insecure permissions".to_string(),
93            ));
94        }
95    }
96
97    let CONTENT = fs::read_to_string(CONFIG_PATH).map_err(|E| {
98        ServiceError::Config(format!("Failed to read config file {CONFIG_PATH}: {E}"))
99    })?;
100
101    Ok(CONTENT)
102}

104/// Loads and validates configuration from system directories.
105///
106/// # Errors
107/// Returns `ServiceError::Config` if:
108/// - No valid config file is found in system directories
109/// - Config file has invalid permissions or format
110/// - Configuration values fail validation checks
111pub fn load_config() -> Result<Config, ServiceError> {
112    let FILENAME = crate::service::config_service::get_config_file_name();
113    let CONTENT = load_config_by_filename(FILENAME)?;
114
115    let CONFIG: Config = toml::from_str(&CONTENT)
116        .map_err(|e| ServiceError::Config(format!("Invalid configuration format: {e}")))?;
117
118    // Validate all fields using configurable limits
119    validate_all_config_fields(&CONFIG)?;
120
121    if !CONFIG
122        .SERVICE_NAME
123        .chars()
124        .all(|c| c.is_alphanumeric() || c == '-' || c == '_')
125    {
126        return Err(ServiceError::Config(
127            "Invalid service name characters".to_string(),
128        ));
129    }
130
131    let LOG_PATH = Path::new(&CONFIG.LOG_FILE_PATH);
132    if !LOG_PATH.is_absolute() {
133        return Err(ServiceError::Config(
134            "Log path must be absolute".to_string(),
135        ));
136    }
137
138    // Basic path validation - detailed security checks happen at use time
139    let ALLOWED_PREFIXES = ["/var/log", "/opt"];
140    if !ALLOWED_PREFIXES
141        .iter()
142        .any(|prefix| LOG_PATH.starts_with(prefix))
143    {
144        return Err(ServiceError::Config(
145            "Log path not in allowed directory".to_string(),
146        ));
147    }
148
149    // Validate install and config directories
150    let INSTALL_PATH = Path::new(&CONFIG.INSTALL_DIR);
151    let CONFIG_PATH_DIR = Path::new(&CONFIG.CONFIG_DIR);
152
153    if !INSTALL_PATH.is_absolute() || !CONFIG_PATH_DIR.is_absolute() {
154        return Err(ServiceError::Config(
155            "Install and config paths must be absolute".to_string(),
156        ));
157    }
158
159    Ok(CONFIG)
160}
