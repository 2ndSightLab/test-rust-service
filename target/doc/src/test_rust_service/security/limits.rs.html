1use crate::service::error::ServiceError;
2
3pub fn get_file_descriptor_limit() -> Result<u64, ServiceError> {
4    #[cfg(unix)]
5    {
6        let mut RLIMIT = libc::rlimit {
7            rlim_cur: 0,
8            rlim_max: 0,
9        };
10
11        // SAFETY: getrlimit() is safe when passed a valid resource type and a valid pointer
12        // to a properly initialized rlimit struct. We provide both requirements here.
13        let RESULT = unsafe { libc::getrlimit(libc::RLIMIT_NOFILE, &raw mut RLIMIT) };
14
15        if RESULT != 0 {
16            // Get errno for better error reporting
17            let ERRNO = unsafe { *libc::__errno_location() };
18            return Err(ServiceError::Config(format!(
19                "Failed to get file descriptor limit: errno {ERRNO}"
20            )));
21        }
22
23        // Validate the returned limit is reasonable
24        if RLIMIT.rlim_cur == 0 || RLIMIT.rlim_cur == u64::MAX {
25            return Err(ServiceError::Config(
26                "Invalid file descriptor limit returned by system".to_string(),
27            ));
28        }
29
30        Ok(RLIMIT.rlim_cur)
31    }
32    #[cfg(not(unix))]
33    {
34        Err(ServiceError::Config(
35            "File descriptor limit check not supported on this platform".to_string(),
36        ))
37    }
38}
